Model_data <- read.csv("model_list_20230923.csv")
Proteomics <- read.csv("proteomics_all_20221214.csv")
#View(Model_data)
#View(Proteomics)

Clean up datasets

Proteomics_clean <- Proteomics %>% 
  select(-"uniprot_id", -"model_id")
  
Model_data_clean <- Model_data %>% 
  select("model_name", "tissue", "cancer_type", "cancer_type_detail")

#Merge datasets together
Proteomics_withID <- Proteomics_clean %>% 
  inner_join(Model_data_clean, by = join_by(model_name))

EDA of all proteins

Proteomics_withID %>% 
  ggplot(aes(x = protein_intensity)) +
  geom_density(alpha=0.4, fill="blue") +
  scale_x_continuous(limits = c(0,10)) +
  theme_bw()

Filter for ATF4 Only

#filter proteomics data for ATF4 only
Proteomics_ATF4 <- Proteomics_withID %>% 
  filter(symbol == "ATF4")

EDA of ATF4

#Exploring ATF4 protein levels distribution
Proteomics_ATF4 %>% 
  ggplot(aes(x = protein_intensity)) +
  geom_density(alpha=0.4, fill="blue") +
  scale_x_continuous(limits = c(0,10)) +
  theme_bw()

ATF4 Protein Levels Across Cancer

#Exploring ATF4 protein levels pan-cancer
Proteomics_ATF4_filtered <- Proteomics_ATF4 %>% 
  group_by(cancer_type) %>%
  filter(n() >= 10) %>%
  ungroup() %>% 
  filter(cancer_type != "Other Sarcomas") %>% 
  filter(cancer_type != "Other Solid Cancers")

Proteomics_ATF4_filtered %>%
  ggplot(aes(x=cancer_type, y=protein_intensity, color = tissue)) +
  geom_jitter() +
  geom_boxplot() +
  theme_bw() +
  labs(x = "Cancer Type",
       y = "ATF4 Protein Expression",
       title = "ATF4 protein expression across cancers") +
  theme(axis.text.x = element_text(angle = 90)) +
  theme(legend.position = "none")

ATF4 Protein Levels within SCLC

#Exploring ATF4 protein levels in SCLC
Proteomics_ATF4 %>%
  filter(cancer_type_detail == "Small Cell Lung Carcinoma") %>% 
  ggplot(aes(x = protein_intensity)) +
  geom_density(alpha=0.4, fill="blue") +
  scale_x_continuous(limits = c(0,10)) +
  theme_bw()

ATF4 Protein Levels within Lung Cancers

#Exploring ATF protein levels by subtype in Lung
Proteomics_ATF4 %>% 
  filter(tissue == "Lung") %>% 
   ggplot(aes(x=cancer_type, y=protein_intensity, color = cancer_type)) +
  geom_jitter() +
  geom_boxplot() +
  theme_bw() +
  labs(x = "Lung Cancer Type",
       y = "ATF4 Protein Expression",
       title = "Visualizing ATF4 protein expression across Lung Cancers") +
  theme(axis.text.x = element_text(angle = 90)) +
  theme(legend.position = "none")

Conclusions

ATF4 is expressed ubiquitously across in vitro cancer cell lines. Some lineages particularly blood lineages express ATF4 less than solid cancers.There also seems to be a distribution of ATF4 expression within lineages suggesting heterogeneity within a particular lineage.
