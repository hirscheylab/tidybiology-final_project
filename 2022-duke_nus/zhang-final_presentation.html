EDA

I then did exploratory data analysis with this single cell RNA seq data. First I looked at the abundance of RNA of different transcript length that is specifically expressed in distinct organs from all 10 cells. The transcript length of RNAs that are specifically expressed in brain, heart and kidney is mostly below 20,000 and their quantity is mostly between 50 to 150, whereas the others have more longer transcripts at low quantity.

data("scRNAseq")
pivot_scRNAseq <-
  scRNAseq %>%
  pivot_longer(cols = 14:23, names_to ="cell",
               values_to = "quantity")
pivot_scRNAseq %>%
  ggplot(aes(transcript_length, quantity,color=organ) )+
  geom_point(alpha=0.3)+
  scale_color_manual(values = c("brain" = "cyan",
                                "heart"="magenta",
                                "kidney"="blue",
                                "other"="yellow"))+
  labs(title="transcript_length_quantity")

Next I removed the genes that are expressed in other organs and only look at the abundance of RNA of different transcript length from brain, heart and kidney. The points are colored by their gene type.

In this RNAseq data set, it is shown that brain, heart and kidney-specific genes only have protein_coding and lncRNA genes.

pivot_scRNAseq %>%

To investigate the transcripts that specifically abaundant in brain cells, I reshaped my data and looked at the log2 fold change in the expression level between the pooled data of brain cells relative to other type of cells.

brain_cell_RNAseq <-
  pivot_scRNAseq %>%
  filter(cell=="cell_3"|cell=="cell_5"|cell=="cell_10")
brain_mean_quantity <-
  brain_cell_RNAseq%>%  
  group_by(gene_name) %>% 
  summarise(mean(quantity))%>%
  janitor::clean_names()

other_cell_RNAseq <-
  pivot_scRNAseq %>%
  filter(cell!="cell_3"&cell!="cell_5"&cell!="cell_10")
other_mean_quantity<-
  other_cell_RNAseq%>%  
  group_by(gene_name) %>% 
  summarise(mean(quantity))%>%
  janitor::clean_names()


fold_change_value <- 
  brain_mean_quantity$mean_quantity/other_mean_quantity$mean_quantity
fold_change=tibble(brain_mean_quantity$gene_name,fold_change_value)
fold_change$log2_fold_change=log2(fold_change$fold_change_value)
colnames(fold_change)[1] = "gene_name"

fold_change %>%
  ggplot()+
  geom_col(aes(gene_name, log2_fold_change))+
  labs(title="log2_fold_changes_of_gene_expression_between_brain_cells_and_others")+
   theme(axis.title.x=element_blank(),
        axis.text.x=element_blank(),
        axis.ticks.x=element_blank())

##Final graph

For the final graph, I plotted the -log10 of P-values for comparing the quantity of transcripts between the pooled data of brain cells relative to all other cell types against the log2 of fold changes in the expression levels.

brain_cell_seq2 <-
  scRNAseq %>%
  select (gene_name, cell_3, cell_5, cell_10)

other_cell_seq2 <-
  scRNAseq %>%
  select (gene_name, cell_1, cell_2, cell_4,cell_6, cell_7, cell_8,cell_9)

testresults <- vector("list", nrow(brain_cell_seq2))
pvalues <- vector("numeric", nrow(brain_cell_seq2))
#For loop
for (j in seq(nrow(brain_cell_seq2))){
  testresults[[j]]<- t.test(brain_cell_seq2[, c(2:4)][j,], other_cell_seq2[, c(2:8)][j,])
  pvalues[[j]]<-testresults[[j]]$p.value
}
ggplot()+
   geom_point(data = fold_change, 
              aes(log2_fold_change, log10_pvalue), 
              color="gray") +
  geom_point(data=fold_change %>% filter(log2_fold_change>0 & log10_pvalue>-log10(0.05)), aes(x=log2_fold_change, y=log10_pvalue), color="green") 

interested_genes<-
